        "import numpy as np\n",
        "from scipy.stats import binom\n",
        "\n",
        "# Observed data (heads out of 10 tosses)\n",
        "data = np.array([5, 9, 8, 4, 7])\n",
        "n = 10  # tosses per trial\n",
        "\n",
        "# Initial guesses\n",
        "theta_A = 0.6\n",
        "theta_B = 0.5\n",
        "\n",
        "iterations = 10\n",
        "\n",
        "for i in range(iterations):\n",
        "\n",
        "    # E-Step: compute responsibilities\n",
        "    prob_A = binom.pmf(data, n, theta_A)\n",
        "    prob_B = binom.pmf(data, n, theta_B)\n",
        "\n",
        "    weight_A = prob_A / (prob_A + prob_B)\n",
        "    weight_B = prob_B / (prob_A + prob_B)\n",
        "\n",
        "    # M-Step: update parameters\n",
        "    theta_A = np.sum(weight_A * data) / (np.sum(weight_A) * n)\n",
        "    theta_B = np.sum(weight_B * data) / (np.sum(weight_B) * n)\n",
        "\n",
        "print(\"Estimated theta_A:\", round(theta_A, 3))\n",
        "print(\"Estimated theta_B:\", round(theta_B, 3))"
